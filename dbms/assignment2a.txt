CREATE TABLE employees (
    ID NUMBER PRIMARY KEY,
    Name VARCHAR2(255),
    Age NUMBER,
    Department VARCHAR2(100),
    Salary NUMBER(10, 2),
    Experience NUMBER,
    AreaOfExperties VARCHAR2(255)
);

CREATE OR REPLACE VIEW employee_view AS
SELECT ID, Name, Age, Department, Experience
FROM employees;

CREATE SEQUENCE employee_id_seq START WITH 1 INCREMENT BY 1;

CREATE OR REPLACE TRIGGER employee_id_trg
BEFORE INSERT ON employees
FOR EACH ROW
BEGIN
  :NEW.ID := employee_id_seq.NEXTVAL;
END;
/

CREATE INDEX idx_employee_id ON employees(ID);


CREATE OR REPLACE PUBLIC SYNONYM EMP FOR employees;

SELECT * FROM EMP;



DROP PUBLIC SYNONYM EMP;
DROP TABLE employees CASCADE CONSTRAINTS;
DROP SEQUENCE employee_id_seq;
